    Variable Scope

    There is local scope and global scope. When you declare a variable inside of the if statement then this is not available outside, as it is local to that if statement. If you declare a variable outside of the if statement then you can either change it like this ageOfPerson = 30; - which will change it globally as well, or change it only within the if statement let ageOfPerson = 30;. If you have a further if statement within another, then if you use the variable again it will use the value that is defined in the if statement and not the global variable.

    
let ageOfPerson = 40;

if(true) {
  let ageOfPerson = 30;
  let personsName = 'Chris'; // This is only valid inside this code block, not outside
  // ageOfPerson = 30; then it would reassign the main variable this value too as not using let
  document.write("<p>Inside the if statement the age is - <strong>" + ageOfPerson + "</strong>, and the name is " + personsName + "</p>");

  if(true) {
    document.write("<p>The variable uses the same value within the code block, not the global variable - <  strong>" + ageOfPerson + "</strong></p>");

  • Arrow Functions

    An arrow function is a shorter way of writing a function. You remove the word function followed by => and then if there is one parameter you can remove the return keyword along with the {}. 

    
// regular function
const calculateArea = function(radius) {
  return 3.14 * radius**2;
}

// As only 1 parameter, remove () and remove return
const calculateArea = radius => 3.14 * radius**2;

// If there are two or more parameters
const calculateArea = (width, height) => 3.14 * radius**2;

// If there are NO parameters
const calculateArea = () => 3.14 * radius**2;

    So, taking the above into account, the following are how you would change to an arrow function.

    
// Normal Function
const greet = function() {
  return 'Hello World';
}

// Arrow function
const greet2 = () => 'hello, world';

const greetResult = greet2();
document.write("<p>" + greetResult + "</p>");

  let total = 0;

  return total;

      You can use the ternary operator as the shortcut for the if-else statement as follows:

      
var age = 19;
var canDrive = age > 16 ? 'yes' : 'no';

      In general, the syntax of the ternary operator is as follows:

      
var result = condition ? expression_1 : expression_2;

      The condition is an expression that evaluates to a Boolean value, either true or false. If the condition is true, the ternary operator returns expression_1, otherwise it returns the expression_2.

      It is a best practice to use the ternary operator when it makes the code easier to read. If the logic contains many if...else statements, you shouldn’t use the ternary operators

  var person2 = new person("Holly", 31);

function printFullAge(years) {
  var ages = [];
  var fullAges = [];

  for( var i = 0; i < years.length; i++ ) {
    // Add the age to the years array. 2019 - year of birth will equal the age
    ages[i]=2019 - years[i];
  }

  for(i=0; i <= ages.length - 1; i++) {
    if(ages[i]>= 18) {
      document.write('<p>Person ' + (i + 1) + ' is ' + ages[i] + ' years old, and is of full age</p>');

      fullAges.push(false);
    }
  }

  return fullAges;

}

 // Create the deck and store all cards in deck
function createDeck(){
  let deck = [];
  
    // Cycle through each suit first
    for (let suitIdx = 0; suitIdx < suits.length; suitIdx++ ) {
        // Loop through all of the values
        for (let valueIdx=0; valueIdx < values.length; valueIdx++) {
          // create results into an object
          let card={
          suit: suits[suitIdx],
          value: values[valueIdx]
          }
          // Add the card to the deck
          deck.push(card);
        }
    }
  return deck;
} 
          
// Shuffle the cards, loop through all cards in the deck and swap cards so random 

function shuffleDeck(deck) {
  for (let i=0; i < deck.length; i++) {
  // Math.random * 52 will make a number between 1 and 51, then trunc will make sure that the number is not a decimal
    let swapIdx=Math.trunc(Math.random() * deck.length);
    let tmp=deck[swapIdx];
    deck[swapIdx]=deck[i];
    deck[i]=tmp;
    }
}
